let thirst=document.querySelector('.thirst-child')
thirst.addEventListener('click', click)
function click(event){
 console.log('Последний элемент отработал')
  setTimeout(function(){
  thirst.style.backgroundColor='purple'
},2000)
setTimeout(function(){
  thirst.style.backgroundColor='brown'
},3000)


}
